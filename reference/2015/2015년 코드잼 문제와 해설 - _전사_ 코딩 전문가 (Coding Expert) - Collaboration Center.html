#include <stdio.h>
#include <limits.h>
#include <stdlib.h>
#include <vector>
#include <algorithm>
 
using namespace std;
 
int gN;
char gP[257];
  
int gPos[][2] = {
    { 1, 3 },   // 0
    { 0, 0 },   // 1
    { 1, 0 },   // 2
    { 2, 0 },   // 3
    { 0, 1 },   // 4
    { 1, 1 },   // 5
    { 2, 1 },   // 6
    { 0, 2 },   // 7
    { 1, 2 },   // 8
    { 2, 2 },   // 9
};
  
#define GETX(ch)    (gPos[(ch) - '0'][0])
#define GETY(ch)    (gPos[(ch) - '0'][1])
  
int calc(char s[]) {
    int res = 0;
    int dx = 0, dy = 0;
  
    if (s[1] != s[0]) {
        dx = GETX(s[1]) - GETX(s[0]);
        dy = GETY(s[1]) - GETY(s[0]);
        res += (abs(dx) <= 1 && abs(dy) <= 1) ? 1 : 2;
    }
  
    for (int i = 2; s[i]; i++) {
        if (s[i] == s[i - 1]) {
            dx = 0;
            dy = 0;
            continue;
        }
        int tdx = GETX(s[i]) - GETX(s[i - 1]);
        int tdy = GETY(s[i]) - GETY(s[i - 1]);
        if (abs(tdx) <= 1 && abs(tdy) <= 1) {
            if (dx == tdx && dy == tdy) {
                res += 1;
            } else {
                res += 2;
            }
        } else {
            res += 3;
        }
 
        dx = tdx;
        dy = tdy;
    }
 
    return res;
}
  
int main(int argc, char* argv[]) {
    FILE* pfIn = fopen(argv[1], "rt");
    FILE* pfOut = fopen(argv[2], "wt");
  
    int T;
    fscanf(pfIn, "%d", &T);
    while (T-- > 0) {
        fscanf(pfIn, "%d", &gN);
 
        int minV;
        char minP[257];
        fscanf(pfIn, "%s", minP);
        minV = calc(minP);
        for (int i = 1; i < gN; i++) {
            fscanf(pfIn, "%s", gP);
            int v = calc(gP);
            if (v < minV) {
                strcpy(minP, gP);
                minV = v;
            }
        }
        fprintf(pfOut, "%s\n", minP);
    }
  
    fclose(pfIn);
    fclose(pfOut);
 
    return 0;
}

해설 :

2번 문제는 그리 어렵지는 않은데 문제에 주어진 조건을 이해하는 과정에서 많이들 실수를 한 것 같습니다.

먼저 흔적을 빼는 경우도 없다고 가정하고 고대 바빌로니아 성벽일 수 있는지를 판단하는 방법에 대해서 생각해보겠습니다.
이미 대부분 알고 계시겠지만 가장 바깥쪽 흔적들을 연결하는 사각형에서 각 변에서 안쪽으로 W만큼 들어간 사각형을 만들었을 때, 그 안에 흔적이 있으면 고대 바빌로니아 성벽일 수 없습니다.
예를 들어 아래와 같은 흔적이 주어졌고, W가 1이라면 안쪽 영역에 흔적이 1개 있으므로 고대 바빌로니아 성벽일 수 없습니다.

그런데 주어지는 입력 중에는 이 안쪽 영역이 안만들어지는 것들이 있습니다. 이건 입력이 잘못된걸까요?
예를 들어서 위 예에서와 같이 흔적이 주어지고 W가 3이 주어졌다고 합시다.
바깥 사각형의 테두리가 가로/세로 5인데, 벽 두께 6(W * 2)보다도 작습니다. 당연히 안쪽 사각형이 안만들어지죠

여기서 다시 문제를 곱씹어볼 필요가 있습니다.
분명 문제에서는 “흔적의 일부를 발견했다”라고 하고 있습니다. 즉, 위 사각형 경계 바깥으로도 성벽이 있을 수 있다는 것입니다.
다시 말해서 아래와 같이 성벽이 구성될 수도 있는 것이죠.

이렇게 만들고 보면 고대 바빌로니아 성벽이 될 수도 있다는 것을 알 수 있습니다.
즉, 안쪽 영역의 가로 또는 세로 크기가 1이하인 경우(1이거나 영역이 없는 경우)는 무조건 고대 바빌로니아 성벽이 될 수 있습니다.
이것을 조건식으로 만들면 다음과 같이 됩니다.

if ((maxX – minX <= W * 2 + 1) || (maxY – minY <= W * 2 + 1))
고대 바빌로니아 성벽일 수 있음

안쪽 영역 검사하기 전에 이 조건을 먼저 검사해서 영역이 생성되지 않는 경우를 걸러내면 좀 더 명확하게 문제를 풀어나갈 수 있습니다.

이제 흔적을 빼는 것을 고려해보겠습니다.
많은 분들이 네 모서리의 흔적을 뺄 수 있는지, 뺐을 때 왼쪽 또는 오른쪽 경계가 변해야 하는지에 대해서 혼란스러워 했던 것으로 압니다.
아래와 같은 흔적들이 주어졌다고 하고, 빨간색 흔적을 지울 수 있는지, 지운다고 했을 때 왼쪽 경계가 바뀌어야 하는지 알아보도록 하겠습니다.

결론을 먼저 말씀드리면 빨간색 흔적도 뺄 수 있습니다. 이유는 간단합니다. 북쪽으로 무너져서 흩어진 성벽의 흔적으로도 볼 수 있기 때문입니다.
문제는 ‘뺐을 때 좌측 경계가 바뀌어야 하는 것인가’ 인데요. 바뀔 수 없습니다.
이유는 우리에게 주어지는 입력은 성벽의 흔적이고, 문제에서 뺄 수 있는 흔적은 성벽의 북쪽과 남쪽으로만 무너진 흔적이라고 했기 때문입니다.
다시 말해서 빨간색 흔적을 뺀다는 것은 빨간색 흔적 밑 어느 부분(X표시한 부분)인가에 진짜 성벽이 있다는 것을 말하는 것이므로 왼쪽 경계는 변할 수 없게 됩니다.

반대로 안뺀다면 빨간색 흔적이 진짜 성벽의 흔적이라는 얘기이므로 당연히 왼쪽 경계는 변경할 수 없을 거구요.

즉, 흔적을 빼는 것이 좌/우측 경계에는 영향을 미치지 않으며, 흔적을 빼는 것도 위쪽과 아래쪽에서만 할 수 있습니다.

실제 흔적을 빼는 것은 하나씩 빼보면 됩니다.
남쪽에서 0개를 빼고 북쪽에서 최대 K개를 뺀 후 확인하고, 또 남쪽에서 1개를 빼고 북쪽에서 K – 1개를 뺀 후 확인하고, ...
이런 식으로 모든 경우에 대해서 확인해보면 됩니다.

이러한 내용을 기반으로 다음과 같은 알고리즘을 만들 수 있습니다.

1) 흔적 위치를 입력 받아 Y축 기준으로 정렬한다.
2) minX와 maxX를 구하고 (maxX – minX) <= W * 2 + 1 이면 “YES” 출력하고 종료한다.
3) 남쪽 흔적을 0에서 K개까지 차례대로 빼면서 아래 테스트를 진행한다.
i) 남쪽에서 흔적을 a개 뺐다면, 북쪽에서 K – a 개를 뺀 후 minY, maxY를 구한다.
ii) 사각형 (minX + W, minY + W) – (maxX – W, maxY – W) 안쪽에 흔적이 있는지 보고,
흔적이 없다면 고대 바빌로니아 흔적일 수 있으므로 “YES”를 출력한 후 종료한다.
4) 고대 바빌로니아 흔적이 아니므로 “NO”를 출력한다.

#include <stdio.h>
#include <limits.h>
#include <stdlib.h>
#include <vector>
#include <algorithm>
 
using namespace std;
  
int gN;
int gK;
int gW;
 
struct Pt {
    int x;
    int y;
 
    bool operator <(const Pt& r) const {
        if (y < r.y)
            return true;
        else if (y > r.y)
            return false;
        else
            return x < r.x;
    }
};
Pt gP[300001];
 
bool solve() {
    int minx, maxx;
    int miny, maxy;
 
    minx = INT_MAX;
    maxx = INT_MIN;
    for (int i = 0; i < gN; i++) {
        if (minx > gP[i].x)
            minx = gP[i].x;
        if (maxx < gP[i].x)
            maxx = gP[i].x;
    }
    minx += gW;
    maxx -= gW;
    if (maxx - minx <= 1)
        return true;
 
    Pt pt;
    int ys, ye, yt;
 
    ys = 0;
    do {
        ye = gN - (gK - ys);
        if (ye <= ys)
            return true;
 
        miny = gP[ys].y + gW;
        maxy = gP[ye - 1].y - gW;
        if (maxy - miny <= 1)
            return true;
 
        pt.x = INT_MAX;
        pt.y = miny;
        int lo = upper_bound(gP + ys, gP + gN, pt) - gP;
        pt.x = INT_MIN;
        pt.y = maxy;
        int hi = upper_bound(gP + ys, gP + gN, pt) - gP;
 
        if (lo >= hi)
            return true;
 
        bool ok = true;
        for (int i = lo; i < hi; i++) {
            if (minx < gP[i].x && gP[i].x < maxx) {
                ok = false;
                break;
            }
        }
        if (ok)
            return true;
 
        // move to next
        yt = gP[ys++].y;
        while (ys <= gK && gP[ys].y == yt)
            ys++;
    } while (ys <= gK && ys < gN);
 
    return false;
}
 
int main(int argc, char* argv[]) {
    FILE* pfIn = fopen(argv[1], "rt");
    FILE* pfOut = fopen(argv[2], "wt");
 
    int T;
    fscanf(pfIn, "%d", &T);
    while (T-- > 0) {
        fscanf(pfIn, "%d %d %d", &gN, &gK, &gW);
        for (int i = 0; i < gN; i++)
            fscanf(pfIn, "%d %d", &gP[i].x, &gP[i].y);
        sort(gP, gP + gN);
 
        fprintf(pfOut, "%s\n", solve() ? "YES" : "NO");
    }
  
    fclose(pfIn);
    fclose(pfOut);
 
    return 0;
}

DP 문제 중에서도 꽤 어려운 문제였습니다.
풀이 방법은 의외로 다양합니다만, 기본 원리는 같습니다.
추 후 원래 출제하셨던 분들의 제대로된 풀이가 올라오면 아시게되겠지만 풀이방법이 1가지만 있는게 아니며,
여기서는 그 중 1가지 방법에 대해서만 소개해드리겠습니다.

주어진 조건을 조금 다르게 해석해보도록 하겠습니다.
어떤 재배치한 결과가 전체 N개인 입력 I에서 앞에서 n개를 빼고 뒤에서 m개를 빼서 만들어졌다고 한다면,
아래와 같은 배열 A, B를 순서를 유지하면서 섞은 것 중 하나가 됩니다.

A(n)={ I[0], I[1], …, I[n-1] }, B(m)={ I[N-1], I[N-2], …, I[N-m] }

이것을 조금 확장하고 일반화 하여 A(i)와 B(j)를 섞는 방법 중에서 거리의 합이 가장 작은 값을 D(i, j)라고 한다면, D(i, j)는 다음과 같은 2차원 배열로 표현할 수 있습니다.

결국 답은 아래의 경우 중 최소값이 됩니다.

- D(0, N) = A(0), B(N)을 섞는 방법 중에서 거리의 합이 가장 작은 값
- D(1, N-1) = A(1), B(N-1) 을 섞는 방법 중에서 거리의 합이 가장 작은 값
- D(2, N-2) = A(2), B(N-2) 을 섞는 방법 중에서 거리의 합이 가장 작은 값
. . .
- D(N-1, 1) = A(N-1), B(1) 을 섞는 방법 중에서 거리의 합이 가장 작은 값
- D(N, 0) = A(N), B(0) 을 섞는 방법 중에서 거리의 합이 가장 작은 값

D 배열에서 이 값들은 아래 노란색으로 표시된 영역을 나타내고, 회색 영역은 의미 없는 영역입니다.

D(0,0)부터 시작해서 아래 방향으로 차례대로 배열 D를 계산해나가면 답을 찾을 수 있습니다.

그런데 여기까지 생각했다고 하더라도, D(i, j)의 값을 어떻게 구해야하는 지를 모르면 풀 수가 없습니다.
D(i, j)는 다음과 같이 구할 수 있습니다.

D(i, j) = min(D(i – 1, j)를 구성하는 재배치 숫자 순서에 앞에서 꺼낸 I[i-1]를 뒤에 붙였을 때의 거리의 합,
D(i, j-1)를 구성하는 재배치 숫자 순서에 뒤에서 꺼낸 I[N-j]를 뒤에 붙였을 때의 거리의 합

여기서 (i, j)는 최종 종점이 아니기 때문에 실제 거리의 총합이 아니고, D(i, j)를 구성하는 데까지의 거리의 합이면 됩니다.
예를 들어서 아래와 같이 D(2, 3)을 구성하는 경우가 아래와 같다면, D(2, 3) = 11 이 됩니다.

결국 D(i, j)를 구성하는 재배치 숫자 순서에 다른 숫자 a를 추가할 때의 거리의 합은 다음과 같이 되는데,

D(i, j) + ‘a가 추가된 후에도 연결되지 않는 edge의 수’

여기서 ‘a가 추가된 후에도 연결되지 않는 edge의 수’는 다시 다음과 같이 계산할 수 있습니다.

‘D(i, j) 구성 숫자 들에서 연결되지 않은 edge의 수’(E(i, j)) – ‘a가 추가되면서 연결되는 edge의 수’(CE(a))
+ ‘a가 추가된 후에도 연결되지 않는 a의 edge 수’(UE(a))

이상을 종합하면 D(i, j)는 다음과 같이 계산할 수 있으며, 이제 이 값들을 사용하여 DP(Dynamic Programming)나 메모이제이션으로 답을 찾을 수 있습니다.

D(i, j) = min(D(i – 1, j) + E(i-1, j) – CE(I[i-1]) + UE(I[i-1]),
D(i, j-1) + E(i, j-1) – CE(I[N-j]) + UE(I[N-j]))

E(i, j) = E(i-1, j) – CE(I[i-1]) + UE(I[i-1]) 와 E(i, j-1) – CE(I[N-j]) + UE(I[N-j]) 중 D(i, j)를 최소로 만드는 경우의 값

 

여기서는 DP로 푸는 방법을 소개합니다.

    int offset;
  
    if (gNodes[start].depth >= gNodes[end].depth)

        s = gNodes[s].parent;

Optimal solution

위에서 제시한 방법으로 large set을 돌리면 시간이 매우 오래 걸립니다.
제한시간 내에 답이 나오기 때문에 pass하는 데에는 문제가 없겠지만, 좀더 빠른 방법이 있습니다.
지금부터 소개해드릴 방법으로 하면 2~3초 정도의 시간에 4번 large set 답이 나옵니다. (C++ 기준)

위에서 구현한 내용을 가지고 profiling해보면 가장 시간이 오래 걸리는 부분은 공통 조상을 찾는 과정입니다.
부모를 하나씩 찾아 올라가야 하기 때문에 한 쌍의 노드를 조사하는데 O(N), K개의 쌍을 조사해야 하므로 O(NK)의 시간이 소요되며, N과 K가 늘어날수록 엄청나게 느려지게 됩니다.
기본 원리와 동작 방식은 그대로 두고 이 탐색 시간만 줄이면 엄청난 성능 향상이 있을 수 있습니다.

지금부터 O(N)의 공통조상 탐색 시간을 O(logN)으로 줄일 수 있는 방법에 대해서 알아보겠습니다.
먼저 각 노드들에 대해 아래와 같은 형태의 테이블을 만듭니다.
예를 들어 아래 A라는 숫자의 노드가 있다고 합시다. 그럼 A숫자 노드의 테이블을 다음과 같이 만듭니다.

이 테이블은 모든 노드에 대해서 만들어야 하며, 이 테이블을 만드는 과정을 pseudo 코드로 정리하면 다음과 같습니다.

 

for (int node = 1; node <= N; node++)
table[node][0] = parent[node];

for (int i = 1; i < LOG; i++) {
for (int node = 1; node <= N; node++) {
int prev = table[node][i - 1];
table[node][i] = table[prev][i - 1];
}
}

 

일단 이렇게 모든 노드에 대해서 table을 만들고 나면 그 다음부터는 이 테이블을 활용해서 O(logN) 시간 안에 특정 거리의 조상으로 갈 수 있습니다. 즉, 일단 기본 solution에서 보여드렸던 step 1단계를 아래와 같은 함수를 통해 O(logN) 시간에 수행 가능합니다.

 

// input: node - 노드 번호, distance - 올라갈 거리
// return: node의 distance 만큼 올라간 조상 노드 번호
int climbTree(int node, int distance) {
int i = 0;
while (distance > 0) {
if (distance & 1)
node = table[node][i];
distance >>= 1;
i++;
}
return node;
}

 

즉, 각 노드에서 2^0, 2^1, 2^2, …, 2^n 올라간 위치의 조상 노드를 구했으므로, table을 이용하여 거리(distance)의 각 비트에 해당하는 만큼 건너뛰면서 조상으로 거슬러 올라갈 수 있습니다.

그 다음 step2 단계는 table을 적극 활용합니다. 다음과 같이 같은 depth는 갖는 A와 B 노드가 있다고 합시다.
그럼 table[A]와 table[B]의 0 ~ 2번까지는 다르지만 3번부터는 같은 값을 갖게 됩니다.
즉, 2번과 3번 사이에 최초로 만나는 공통 조상이 있다는 뜻이죠.

이쯤 되면 최초로 만나는 공통 조상(Least Common Ancestor)은 어떻게 구하는지 알겠죠?
table[A][2]와 table[A][3], 그리고 table[B][2]와 table[B][3] 사이에서 binary search를 하면 최초로 같아지는 조상 노드를 찾으면 돕니다.
이것도 O(logN) 시간에 구할 수 있습니다.

int loA = table[A][2];
int hiA = table[A][3];
int loB = table[B][2];
int hiB = table[B][3];

while (depth[loA] > depth[hiA]) {
int midDepth = (depth[loA] + depth[hiA] + 1) / 2;

int midA = climbTree(loA, depth[loA] - midDepth);
int midB = climbTree(loB, depth[loB] - midDepth);

if (midA == midB) {
hiA = midA;
hiB = midB;
}
else {
loA = parent[midA];
loB = parent[midB];
}
}
// loA가 LCA가 됨

이렇게 공통 조상을 찾았으면 공통 조상과 입력으로 주어지는 두 노드(문제에서 주어지는 두 방의 번호) 사이의 거리를 알 수 있습니다.
이 거리의 합의 1/2 만큼을 입력으로 주어지는 두 노드 중 depth가 큰 노드에서부터 올라가면 우리가 찾고자 하는 노드가 됩니다.

dist = (depth[nodeA] - depth[LCA] + depth[nodeB] - depth[LCA]) / 2;
climbTree(nodeA, dist);

    if (distance <= 0)
        return node;
 
    if (distance >= gNodes[node].depth)
        return 1;

        if (distance & 1)
            node = gAncestorTable[node][i];
        distance >>= 1;
    }

    if (gNodes[start].depth < gNodes[end].depth) {
        swap(start, end);
        swapped = true;
    }
 
    int s = climbTree(start, gNodes[start].depth - gNodes[end].depth);
 
    int upDist = 0;
 
    if (s == end) {
        upDist = (gNodes[start].depth - gNodes[end].depth + swapped) / 2;
    } else {
        int i;
        for (i = 0; i < LOG; i++) {
            if (gAncestorTable[s][i] == gAncestorTable[end][i])
                break;
        }
        if (i == 0) {
            s = gAncestorTable[s][0];
        } else {
            int loS = gAncestorTable[s][i - 1];
            int hiS = gAncestorTable[s][i];

            int hiE = gAncestorTable[end][i];

                int midE = climbTree(loE, gNodes[loE].depth - midDepth);

                    hiS = midS;
                    hiE = midE;
                } else {
                    loS = gNodes[midS].parent;
                    loE = gNodes[midE].parent;
                }
            }
            s = loS;
        }
        upDist = (gNodes[start].depth - gNodes[s].depth +
                  gNodes[end].depth - gNodes[s].depth + swapped) / 2;
    }
 
    return climbTree(start, upDist);
}
