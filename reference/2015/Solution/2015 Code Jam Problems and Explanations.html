#define GETX(ch)    (gPos[(ch) - '0'][0])
#define GETY(ch)    (gPos[(ch) - '0'][1])
  
int calc(char s[]) {
    int res = 0;
    int dx = 0, dy = 0;
  
    if (s[1] != s[0]) {
        dx = GETX(s[1]) - GETX(s[0]);
        dy = GETY(s[1]) - GETY(s[0]);
        res += (abs(dx) <= 1 && abs(dy) <= 1) ? 1 : 2;
    }
  
    for (int i = 2; s[i]; i++) {
        if (s[i] == s[i - 1]) {
            dx = 0;
            dy = 0;
            continue;
        }
        int tdx = GETX(s[i]) - GETX(s[i - 1]);
        int tdy = GETY(s[i]) - GETY(s[i - 1]);
        if (abs(tdx) <= 1 && abs(tdy) <= 1) {
            if (dx == tdx && dy == tdy) {
                res += 1;
            } else {
                res += 2;
            }
        } else {
            res += 3;
        }
 
        dx = tdx;
        dy = tdy;
    }
 
    return res;
}

#include <stdlib.h>
#include <vector>
#include <algorithm>
  
using namespace std;
  
#define MAX_N   11
  
int gN;
int gM;
  
vector<int> gEdge[MAX_N];
  
int gPath[MAX_N];
int gNumToIndex[MAX_N];
  
void makePath(int path, int out[]) {
    int f = 1, l = gN;
    int mask = 1 << gN;
    for (int i = 0; i < gN; i++) {
        mask >>= 1;
        if (path & mask)
            out[i] = l--;
        else
            out[i] = f++;
    }
}
  
int calcDist(int path) {
    // 2^N 가지 중 하나의 path에 대한 숫자 sequence를 구한다
    makePath(path, gPath);
  
    for (int i = 0; i < MAX_N; i++)
        gNumToIndex[i] = -1;
  
    // path에 대한 거리의 합을 구한다.
    int res = 0;
    for (int i = 0; i < gN; i++) {
        int num = gPath[i];
  
        gNumToIndex[num] = i;
        for (int j = 0; j < gEdge[num].size(); j++) {
            int y = gEdge[num][j];
            if (gNumToIndex[y] >= 0)
                res += abs(i - gNumToIndex[y]);
        }
    }
  
    return res;
}
  
int solve() {
    int maxN = 1 << gN;
  
    int minDist = INT_MAX;
    int minPath = 0;
  
    // 0에서 2^N - 1 까지 반복한다.
    for (int i = 0; i < maxN; i++) {
        int v = calcDist(i);
        if (v < minDist) {
            minDist = v;
            minPath = i;
        }
    }
  
    return minDist;
}

{
