2nd LGE Code Jam 2018 - Problem D (English)

Skip to end of metadata
  • Created by 안연하 yeonha.an, last modified by 이주성 jooseong.lee on 2018/08/30
Go to start of metadata

City Planning


Subtask 1

First, we can observe that a number of city blocks with parks built should be even,

so it suffices to check whether city blocks can be entirely filled with 2 x 1 park grids for at most 13 Z-values.

Note that the detailed park grid configuration is not important, because park grid cost per block is fixed.

Bruteforce is enough to solve Subtask 1.


Complexity: < 13 x 4^(12)


Subtask 2

An improved approach - better than bruteforce check - is needed to solve Subtask 2.


Now we color city block grid with black and white so that adjacent grids have different colors.

Then each of 2 x 1 park tiles must be composed of a single white grid + single black grid.

Hence the original problem converts into the problem to check

whether domino tiling is possible for an "holed" (grids that skyscrapers are built on) board.


We can represent this with perfect bipartite matching problem on bipartite graph with

set of white grids(vertices) and black grids and edges representing adjacency.

The well known algorithms like Ford-Fulkerson (O(EV)), Edmonds-Karp are enough to solve Subtask 2.


Subtask 3

We need further improvement to solve Subtask 3 as the number of vertex and edges are significantly increased.

Hopcroft-Karp algorithm (Complexity O(E√V), https://www.geeksforgeeks.org/hopcroft-karp-algorithm-for-maximum-matching-set-1-introduction/) can be applied,

and we can think additional adjustments.


1) First, we can filter the possible Z-values with only checking equality of numbers of remaining white/black grids,

and compute the total budget values corresponding to each possible Z-values before actual testing on perfect bipartite matching.

(The total cost is Z * W + (X / 2) * D where X is number of remaining grids after building skyscrapers)


2) If we found that the perfect matching is actually possible on the Z-value that yields the minimum cost, then we don't need to test remaining Z-values.

So we could sort the possible Z-values in ascending order by the total cost (NOT by Z-value itself)

and then test the perfect matching from the minimum-cost-yielding Z-value to reduce time.


In addition, we must use long type because the budget value exceeds the range of int type.



Implementation (Java):

2nd LGE CodeJam 2018, Problem 4

import java.util.*;
import java.util.Map.Entry;
import java.io.*;
 
public class Main {
 
    public static int R;
    public static int C;
    public static long D;
    public static long W;
    public static int V;
    public static int color[]; // +1 white, -1 black, 0 hole
    public static List<Integer>[] adj; // white -> black adj
 
    public static final int dx[] = { -1, 1, 0, 0 };
    public static final int dy[] = { 0, 0, 1, -1 };
 
    public static Map<Long, Long> prices = new HashMap<Long, Long>();
 
    public static boolean inRange(int y, int x) {
        if (x >= 0 && x < C && y >= 0 && y < R) {
            return true;
        }
        return false;
    }
 
    public static int whiteSize;
    public static int blackSize;
 
    public static void removeVertex(int index, List<Integer>[] clonedAdj) {
        if (color[index] != 0) {
            clonedAdj[index].clear();
            int y = index / C;
            int x = index % C;
            for (int k = 0; k < 4; k++) {
                int y2 = y + dy[k];
                int x2 = x + dx[k];
                if (inRange(y2, x2)) {
                    int index2 = y2 * C + x2;
                    if (color[index2] != 0) {
                        clonedAdj[index2].remove((Integer) index);
                    }
                }
            }
            if (color[index] == 1) {
                whiteSize--;
            } else {
                blackSize--;
            }
        }
        color[index] = 0;
    }
 
    public static void findPossiblePrices(List<Long> Yvalues, Map<Long, ArrayList<Integer>> Ymap, Long baseValue) {
        int graphSize = whiteSize + blackSize;
        if (graphSize % 2 == 0) {
            Long value = (graphSize / 2) * D;
            if (value < baseValue) {
                prices.put(0L, value);
            }
        }
        for (Long Z : Yvalues) {
            graphSize -= Ymap.get(Z).size();
            if (graphSize % 2 == 0) {
                Long value = Z * W + (graphSize / 2) * D;
                if (value < baseValue) {
                    prices.put(Z, value);
                }
            }
        }
    }
 
    public static List<Long> getKeysSortedByPrice(Map<Long, Long> map) {
        List<Entry<Long, Long>> list = new ArrayList<>(map.entrySet());
        list.sort(Entry.comparingByValue());
        List<Long> result = new ArrayList<Long>();
        for (Entry<Long, Long> entry : list) {
            result.add(entry.getKey());
        }
        return result;
    }
 
    public static boolean VerifyMatching(Long Z, List<Long> Yvalues, Map<Long, ArrayList<Integer>> Ymap) {
        List<Integer>[] clonedAdj = new List[V];
        for (int i = 0; i < V; i++) {
            clonedAdj[i] = new ArrayList<Integer>(adj[i]);
        }
        if (Z == 0L) {
            if (whiteSize != blackSize) {
                return false;
            }
            int parkCount = maximumMatching(clonedAdj);
            if (parkCount != whiteSize) {
                return false;
            } else {
                return true;
            }
        }
 
        for (Long Y : Yvalues) {
            for (Integer index : Ymap.get(Y)) {
                removeVertex(index, clonedAdj);
            }
            if (Y == Z) {
                break;
            }
        }
        if (whiteSize != blackSize) {
            return false;
        }
        int parkCount = maximumMatching(clonedAdj);
        if (parkCount != whiteSize) {
            return false;
        } else {
            return true;
        }
    }
 
    // Hopcroft-Karp
    public static int maximumMatching(List<Integer>[] clonedAdj) {
        int cardinality = 0;
 
        int[] mate = new int[V]; // mate[v] = w if v->w is in current matching,
                                    // -1 otherwise
        for (int i = 0; i < V; i++) {
            mate[i] = -1;
        }
        int[] level = new int[V]; // distance from starting path
 
        while (hasAugmentingPath(mate, level, clonedAdj)) { // if augmenting path found, extend matching
            List<Integer>[] adjCopy = new List[V];
            for (int i = 0; i < V; i++) {
                adjCopy[i] = new ArrayList<Integer>(clonedAdj[i]);
            }
            for (int s = 0; s < V; s++) {
                if (mate[s] != -1 || color[s] != 1) {
                    continue;
                }
                Stack<Integer> path = new Stack<Integer>();
                path.push(s);
                while (!path.isEmpty()) {
                    int v = path.peek();
                    int size = adjCopy[v].size();
                    if (size == 0) { // no more edges left here, go back
                        path.pop();
                    } else { // go straight
                        int w = adjCopy[v].get(size - 1);
                        adjCopy[v].remove(size - 1);
                        // go v->w only if v->w is in level graph
                        if ((level[w] == level[v] + 1)
                                && ((color[v] == 1 && mate[v] != w) || (color[v] == -1 && mate[v] == w))) {
                            path.push(w);
 
                            if (mate[w] == -1) {
                                // update matching
                                while (!path.isEmpty()) {
                                    int x = path.pop();
                                    int y = path.pop();
                                    mate[x] = y;
                                    mate[y] = x;
                                }
                                cardinality++;
                            }
 
                        } else {
                            continue;
                        }
 
                    }
                }
            }
        }
 
        return cardinality;
    }
 
    public static boolean hasAugmentingPath(int[] mate, int[] level, List<Integer>[] clonedAdj) {
        boolean found = false;
 
        Queue<Integer> queue = new PriorityQueue<Integer>();
 
        for (int i = 0; i < V; i++) {
            if (mate[i] == -1 && color[i] == 1) {
                level[i] = 0;
                queue.offer(i);
            } else {
                level[i] = Integer.MAX_VALUE;
            }
        }
 
        boolean[] marked = new boolean[V];
 
        while (!queue.isEmpty()) {
            int v = queue.poll();
            for (Integer w : clonedAdj[v]) {
                // forward edge not in matching or backward edge in matching
                if ((color[v] == 1 && mate[v] != w) || (color[v] == -1 && mate[v] == w)) {
                    if (!marked[w]) {
                        level[w] = level[v] + 1;
                        marked[w] = true;
                        if (mate[w] == -1) {
                            found = true;
                        }
                        if (!found) {
                            queue.offer(w);
                        }
                    }
                }
            }
        }
 
        return found;
    }
 
    @SuppressWarnings("unchecked")
    public static void main(String args[]) throws IOException {
 
        Scanner sc;
 
        sc = new Scanner(System.in);
 
        R = sc.nextInt();
        C = sc.nextInt();
        D = sc.nextLong();
        W = sc.nextLong();
        V = R * C;
        whiteSize = 0;
        blackSize = 0;
        color = new int[V];
        adj = new List[V];
        for (int i = 0; i < V; i++) {
            adj[i] = new ArrayList<Integer>();
        }
 
        Map<Long, ArrayList<Integer>> Ymap = new HashMap<Long, ArrayList<Integer>>();
 
        for (int y = 0; y < R; y++) {
            for (int x = 0; x < C; x++) {
                long value = sc.nextLong();
                if (Ymap.get(value) == null) {
                    Ymap.put(value, new ArrayList<Integer>());
                }
                int index = y * C + x;
                Ymap.get(value).add(index);
                if ((y + x) % 2 == 0) {
                    color[index] = 1; // white
                    for (int k = 0; k < 4; k++) {
                        int y2 = y + dy[k];
                        int x2 = x + dx[k];
                        if (inRange(y2, x2)) {
                            int index2 = y2 * C + x2;
                            adj[index].add(index2);
                            adj[index2].add(index);
                        }
                    }
                    whiteSize++;
                } else {
                    color[index] = -1; // black
                    blackSize++;
                }
            }
        }
 
        Set<Long> Yval = Ymap.keySet();
        List<Long> Yvalues = new ArrayList<Long>(Yval);
        Collections.sort(Yvalues);
 
        // at least one case is possible; all buildings. put this first
        Long largestY = Yvalues.get(Yvalues.size() - 1);
        Long baseValue = largestY * W;
        prices.put(largestY, baseValue);
 
        // first seek possible prices, only checking parity of park grids
        findPossiblePrices(Yvalues, Ymap, baseValue);
        // sort Z values from lowest prices to highest
        List<Long> ZvaluesSortedByPrice = getKeysSortedByPrice(prices);
 
        int initialWhite = whiteSize;
        int initialBlack = blackSize;
        int[] initialColor = new int[V];
        for (int i = 0; i < V; i++) {
            initialColor[i] = color[i];
        }
        // now check whether the Z values give perfect bipartite matching
        for (Long Z : ZvaluesSortedByPrice) {
            if (VerifyMatching(Z, Yvalues, Ymap)) {
                // we've already got the lowest price, just quit
                System.out.println(prices.get(Z));
                break;
            }
 
            whiteSize = initialWhite;
            blackSize = initialBlack;
            for (int i = 0; i < V; i++) {
                color[i] = initialColor[i];
            }
        }
 
        sc.close();
    }
}
